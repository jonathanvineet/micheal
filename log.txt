METHOD 1 (RECOMMENDED): Python (cleanest, safest)
python3 - << 'EOF'
import serial, time

s = serial.Serial(
    "/dev/ttyUSB0",
    115200,
    timeout=2,
    dsrdtr=False,
    rtscts=False
)

time.sleep(2)

def send(cmd):
    print(">>", cmd)
    s.write((cmd + "\n").encode())
    time.sleep(0.3)
    while s.in_waiting:
        print(s.readline().decode(errors="ignore").strip())

send("G91")          # relative mode
send("G1 Z-1 F120")  # move Z down 1mm

s.close()
EOF


✔ This WILL move the Z axis
✔ No printer reset
✔ Proper response handling
